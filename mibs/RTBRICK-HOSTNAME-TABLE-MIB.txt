hostnameConfigTable OBJECT-TYPE
    SYNTAX      SEQUENCE OF HostnameConfigEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
        "A table of sw modue information"
    ::= { hostnameConfig }

hostnameConfigEntry OBJECT-TYPE
    SYNTAX      HostnameConfigEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION

	systemHostname 	    DisplayString,
	rtbrickHostname	    DisplayString
}

systemHostname      OBJECT-TYPE

    STATUS      current
    DESCRIPTION

    ::= { hostnameConfigEntry 1 }

rtbrickHostname      OBJECT-TYPE
    SYNTAX      DisplayString (SIZE (0..32))
    MAX-ACCESS  read-only
    STATUS      current
    DESCRIPTION
	"attribute: rtbrick_hostname (2)"
